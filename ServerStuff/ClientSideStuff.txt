//Raw Commands
CMD			  CODE								CLIENT->SERVER
--------------------------------------------------------------
HELLO 		= 0x00 //Say Hi to the server		oprands()
LOGIN 		= 0x01 //Send Login Details			oprands(user,pass)
REGISTER	= 0x02 //Sends Register Data		oprands(user,pass,displayname,email)
UPDATES 	= 0x03 //Checks for updates			oprands(CID)
DOUPDATES 	= 0x04 //Does the updates			oprands(CID)
LISTR		= 0x05 //Gets List of Rooms			oprands(CID)
JROOM		= 0x06 //Joins a Room				oprands(CID,RID)
CHAT		= 0x07 //Sends Chat					oprands(CID,CID or RID or null_ID) //This command is a bit bloated but it allows for global/room/private chatting
REQUEST		= 0x08 //Sends a friend request		oprands(CID,UID)
LISTF		= 0x09 //Returns a list of friends	oprands(CID)
LISTRF		= 0x0A //List resquests for friends	oprands(CID)
ADDF		= 0x0B //Accept/Deny F Request		oprands(CID,UID,0 or 1)
FORMR		= 0x0C //Creates a Room				oprands(CID,MapData)
IHOST		= 0x0D //Are you host?				oprands(CID)
KICK		= 0x0E //Kicks a player from room	oprands(CID,RID,UID)
INVITEF		= 0x0F //Invite friend to room		oprands(CID,RID,FID)
READY		= 0x10 //Are you ready to play?		oprands(CID,0 or 1 or 2)//1-R 0-NR 2-GR
LEAVER		= 0x11 //Leave the room				oprands(CID,RID)
GRESORCE	= 0x12 //Gets a list of resources	oprands(CID)

CMD			  CODE - (RESPONSE)					SERVER->CLIENT
--------------------------------------------------------------
HELLO 		= 0x00 //A hello from server		returns()
LOGIN 		= 0x01 //Send Login Details			returns(ERR, CID or null)
REGISTER	= 0x02 //Sends Register Data		returns(ERR)
UPDATES 	= 0x03 //Checks for updates			returns(ERR)
DOUPDATES 	= 0x04 //Does the updates			returns(ERR)
LISTR		= 0x05 //Gets List of Rooms			returns(ERR,RID[]) //LEFT OFF HERE

CHAT		= 0x07 //Sends Chat					returns(ERR)
REQUEST		= 0x08 //Sends a friend request		returns(ERR)
LISTF		= 0x09 //Returns a list of friends	returns(ERR,FID[])
LISTRF		= 0x0A //List resquests for friends	returns(ERR,FID[])
ADDF		= 0x0B //Accept/Deny F Request		returns(ERR)
FORMR		= 0x0C //Creates a Room				returns(ERR)
